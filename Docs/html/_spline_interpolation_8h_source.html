225  colvec optimalAlpha(size_t min, size_t max, size_t samples, const Aquila::SignalSource *signal, double widthBetween) {
226 
227  // create t
228  colvec t(4);
229  t[0] = signal->sample(min);
230  t[1] = signal->sample(max);
231 
232  // estimate derivative using forward differences
233  t[2] = (signal->sample(min + 1) - signal->sample(min));
234  t[3] = (signal->sample(max) - signal->sample(max - 1));
235 
236  // generate P
237  mat P(4,POLYNOM_DEGREE);
238  P = fillWithPolynomial(0, widthBetween * min, P);
239  P = fillWithPolynomial(1, widthBetween * max, P);
240  P = fillWithDerivative(2, widthBetween * min, P);
241  P = fillWithDerivative(3, widthBetween * max, P);
242 
243  // calculate pseudo inverse
244  auto alpha = pinv(P) * t;
245 
246  // calculate K
247  mat K(samples, POLYNOM_DEGREE);
248  colvec Y(samples);
249 
250  // iterate and fill
251  for (size_t k = min; k <= max; k++) {
252  K = fillWithPolynomial((int) (k - min), widthBetween * k, K);
253  Y[k - min] = signal->sample(k);
254  }
255 
256  // find nullspace of P and KB Pseudoinverse
257  mat B = null(P, 0);
258  mat KB = K * B;
259  mat KBP = pinv(K * B);
260 
261  // calculate optimal alpha parameters
262  mat BKBP = B * KBP;
263  mat right = K * alpha;
264  mat rightMult = Y - right;
265  mat compRight = BKBP * rightMult;
266  colvec alphaStar = alpha + compRight;
267  return alphaStar;
268  }
