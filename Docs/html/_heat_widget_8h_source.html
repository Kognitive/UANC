76  void configPlot(std::shared_ptr<Aquila::Spectrogram> spectrogramm) {
77 
78  // configure axis rect:// this will also allow rescaling the color scale by dragging/zooming
79  this->xAxis->setLabel("Frames");
80  this->yAxis->setLabel("Frequency");
81 
82  // set up the QCPColorMap:
83  QCPColorMap *colorMap = new QCPColorMap(this->xAxis, this->yAxis);
84  int nx = spectrogramm->getFrameCount();
85  int ny = spectrogramm->getSpectrumSize();
86  colorMap->data()->setSize(nx, ny / 2); // we want the color map to have nx * ny data points
87  colorMap->data()->setRange(QCPRange(0, nx), QCPRange(0, ny / 2)); // and span the coordinate range -4..4 in both key (x) and value (y) dimensions
88 
89  // now we assign some data, by accessing the QCPColorMapData instance of the color map:
90  double rx, ry, rz;
91 
92  //Iterate over all frames
93  for (std::size_t x = 0; x < spectrogramm->getFrameCount(); ++x)
94  {
95  // output only half of the spectrogram, below Nyquist frequency
96  for (std::size_t y = 0; y < spectrogramm->getSpectrumSize() / 2; ++y)
97  {
98  colorMap->data()->cellToCoord(x, y, &rx, &ry);
99  Aquila::ComplexType point = spectrogramm->getPoint(x, y);
100  rz = Aquila::dB(point);
101  colorMap->data()->setCell(x, y, rz);
102  }
103  }
104 
105  // add it to the right of the main axis rect
106  _colorScale->setType(QCPAxis::atRight); // scale shall be vertical bar with tick/axis labels right (actually atRight is already the default)
107  colorMap->setColorScale(_colorScale); // associate the color map with the color scale
108  _colorScale->axis()->setLabel("Spektrogramm");
109 
110  // set the color gradient of the color map to one of the presets:
111  colorMap->setGradient(QCPColorGradient::gpSpectrum);
112  // we could have also created a QCPColorGradient instance and added own colors to
113  // the gradient, see the documentation of QCPColorGradient for what's possible.
114 
115  // rescale the data dimension (color) such that all data points lie in the span visualized by the color gradient:
116  colorMap->rescaleDataRange();
117 
118  // make sure the axis rect and color scale synchronize their bottom and top margins (so they line up):
119  QCPMarginGroup *marginGroup = new QCPMarginGroup(this);
120  this->axisRect()->setMarginGroup(QCP::msBottom|QCP::msTop, marginGroup);
121  _colorScale->setMarginGroup(QCP::msBottom|QCP::msTop, marginGroup);
